\begin{block}{Zeit im Medium $i$\strut}
\[
\only<4>{
t_i
=
\frac{
\sqrt{y_i^2+(x-x_i)^2}
}{
c/n_i
}}
\only<5>{t_i=
\frac{n_i}{c}
\sqrt{y_i^2+(x-x_i)^2}
}
\only<6->{
{\color<7->{red}t}

t_1+t_2
=
\frac{1}{c}\sum_{i=1}^n
n_i\sqrt{y_i^2+(x-x_i)^2}
}
\]
\end{block}}
