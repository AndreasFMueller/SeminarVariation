\begin{column}{0.48\textwidth}
\begin{block}{Integralbedingungen}
\[
\int_{x_1}^{x_2}
G(x,y(x),y'(x))\,dx
=
C
\]
mit ``Gradient''
\begin{gather*}
\frac{\partial L}{\partial y}(x,y(x),y'(x))\qquad\qquad
\\
-\frac{d}{dx}
\frac{\partial L}{\partial y'}(x,y(x),y'(x))
\end{gather*}
\end{block}
\end{column}
\begin{column}{0.48\textwidth}
\begin{block}{Funktionswerte}
\begin{align*}
y(x_0)
&=
\int_{x_1}^{x_0} y'(x) \,dx
\\
&=
\int_{x_1}^{x_2} y'(x)\,dx
\\
&=
\int_{x_1}^{x_2} H(x,y(x),y'(x)) \times
\\
&\qquad\qquad
\vartheta(x_0-x)\,dx
\intertext{mit}
H(x,y,y')
&=
y'\cdot\vartheta(x_0-x)
\end{align*}
\end{block}
\end{column}
\end{columns}
